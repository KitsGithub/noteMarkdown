```js
<template slot="listHeader">
  <cell>
    <shop-detail-card :shop="shop" class="detail-card" @onMapClick="onMapClick()" />
  </cell>
</template>

onMapClick() {
  const data = {
    lat: this.shop.latitude || 0,
    lon: this.shop.longitude || 0,
    poiName: this.shop.name || '',
    type: 0,
  }
  fjkModule.openAmap(data)
},

.detail-card {
  padding-top: 20wx;
  padding-bottom: 20wx;
}
```
